<body>

  <h1>Segnapunti Tressette</h1>


  <div id="setup">

    <label>Numero giocatori:</label>

    <select id="numPlayers">

      <option value="4">4</option>

      <option value="5">5</option>

    </select>

    <div id="playerNames"></div>

    <button onclick="startGame()">Avvia partita</button>

  </div>


  <div id="game" class="hidden">

    <h2>Inserisci punteggi manche</h2>

    <div id="roundInputs"></div>

    <p id="error" class="error"></p>

    <button onclick="confirmRound()">Conferma manche</button>


    <h2>Classifica</h2>

    <ul id="ranking"></ul>

  </div>


  <script>

    let players = [];

    let points = [];


    function startGame() {

      const num = parseInt(document.getElementById("numPlayers").value);

      players = [];

      for (let i = 0; i < num; i++) {

        const name = document.getElementById("name" + i).value || "Giocatore " + (i+1);

        players.push({ name: name, score: 0 });

      }

      points = Array(num).fill(0);

      document.getElementById("setup").classList.add("hidden");

      document.getElementById("game").classList.remove("hidden");

      renderRoundInputs();

      renderRanking();

    }


    function renderPlayerInputs() {

      const num = parseInt(document.getElementById("numPlayers").value);

      const container = document.getElementById("playerNames");

      container.innerHTML = "";

      for (let i = 0; i < num; i++) {

        container.innerHTML += `<input id="name${i}" placeholder="Nome giocatore ${i+1}">`;

      }

    }


    function renderRoundInputs() {

      const container = document.getElementById("roundInputs");

      container.innerHTML = "";

      players.forEach((p, i) => {

        container.innerHTML += `

          <div class="player">

            <label>${p.name}</label>

            <input type="number" id="point${i}" value="0" min="0" max="21">

          </div>`;

      });

    }


    function confirmRound() {

      let total = 0;

      for (let i = 0; i < players.length; i++) {

        points[i] = parseInt(document.getElementById("point"+i).value) || 0;

        total += points[i];

      }

      if (total !== 21) {

        document.getElementById("error").textContent = "La somma deve essere 21 (ora è " + total + ")";

        return;

      }

      document.getElementById("error").textContent = "";

      for (let i = 0; i < players.length; i++) {

        players[i].score += points[i];

      }

      renderRoundInputs();

      renderRanking();

    }


    function renderRanking() {

      const ranking = document.getElementById("ranking");

      ranking.innerHTML = "";

      players

        .slice()

        .sort((a,b) => b.score - a.score)

        .forEach(p => {

          ranking.innerHTML += `<li><span>${p.name}</span><span class="score">${p.score}</span></li>`;

        });

    }


    // inizializza input nomi

    renderPlayerInputs();

    document.getElementById("numPlayers").addEventListener("change", renderPlayerInputs);

  </script>
