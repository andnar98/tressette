        function showToast(message) {

        }


        function startGame() {

            // Get player names

            const numPlayers = parseInt(document.getElementById('numPlayers').value);

            players = [];

            

            for (let i = 1; i <= numPlayers; i++) {

                const name = document.getElementById(`player${i}`).value.trim();

                if (!name) {

                    showToast(`Inserisci il nome del giocatore ${i}`);

                    return;

                }

                players.push({

                    name: name,

                    score: 0

                });

            }


            // Set game mode

                if (player.score >= targetScore) {

                    row.classList.add('winner');

                }


                row.innerHTML = `

                    <div class="player-name">${player.name}</div>

                    <div class="player-score">${player.score}</div>

                `;

                container.appendChild(row);

            });

        }


        function generateMancheInputs() {

            const container = document.getElementById('mancheInputs');

            container.innerHTML = '';

            document.getElementById('mancheNumber').textContent = mancheCount;


            players.forEach((player, index) => {

                const div = document.createElement('div');

                div.className = 'manche-player-input';

                div.innerHTML = `

                    <div class="manche-player-name">${player.name}</div>

                    <input type="number" class="score-input" id="score${index}" min="0" max="${targetScore}" inputmode="numeric">

                `;

                container.appendChild(div);

            });


            // Focus first input

            setTimeout(() => {

                const firstInput = document.getElementById('score0');

                if (firstInput) firstInput.focus();

            }, 100);

        }


        function addManche() {

            const scores = [];

            let total = 0;

            

            // Get scores

            players.forEach((player, index) => {

                const scoreInput = document.getElementById(`score${index}`);

                const score = parseInt(scoreInput.value) || 0;

                scores.push(score);

                total += score;

            });


            // Validate total

            if (total !== targetScore) {

                document.getElementById('scoreError').textContent = 

                    `Errore: il totale deve essere ${targetScore}, ma è ${total}`;

                vibrate();

                return;

            }


            document.getElementById('scoreError').textContent = '';


            // Add scores to players

            players.forEach((player, index) => {

                player.score += scores[index];

            });


            // Add to history

            history.push({

                manche: mancheCount,

                scores: [...scores],

                playerNames: players.map(p => p.name)

            });


            mancheCount++;

            updateScoreboard();

            updateHistory();

            

            // Clear inputs

            players.forEach((player, index) => {

                document.getElementById(`score${index}`).value = '';

            });

            

            document.getElementById('mancheNumber').textContent = mancheCount;

            vibrate();


            // Check for winner

            const winner = players.find(player => player.score >= targetScore);

            if (winner) {

                setTimeout(() => {

                    showToast(`🎉 ${winner.name} ha vinto!`);

                    vibrate();

                }, 500);

            }


            // Focus first input for next round

            setTimeout(() => {

                const firstInput = document.getElementById('score0');

                if (firstInput) firstInput.focus();

            }, 100);

        }


        function updateHistory() {

            const container = document.getElementById('historyList');

            if (history.length === 0) {

                container.innerHTML = '<div style="color: #999; font-style: italic;">Nessuna manche giocata</div>';

                return;

            }


            container.innerHTML = '';

            // Show most recent first

            const reversedHistory = [...history].reverse();

            reversedHistory.forEach(entry => {

                const div = document.createElement('div');

                div.className = 'manche-history';

                const scoresText = entry.scores.map((score, index) => 

                    `${entry.playerNames[index]}: ${score}`

                ).join(', ');

                div.innerHTML = `<strong>Manche ${entry.manche}:</strong> ${scoresText}`;

                container.appendChild(div);

            });

        }


        function undoLastManche() {

            if (history.length === 0) {

                showToast('Nessuna manche da annullare');

                return;

            }


            const lastEntry = history.pop();

            

            // Subtract scores

            players.forEach((player, index) => {

                player.score -= lastEntry.scores[index];

            });


            mancheCount--;

            updateScoreboard();

            updateHistory();

            document.getElementById('mancheNumber').textContent = mancheCount;

            vibrate();

            showToast('Ultima manche annullata');

        }


        function resetGame() {

            if (confirm('Sei sicuro di voler iniziare una nuova partita?')) {

                document.getElementById('gameBoard').style.display = 'none';

                document.getElementById('setupForm').style.display = 'block';

                

                // Reset variables

                players = [];

                mancheCount = 1;

                history = [];

                

                // Clear inputs

                const numPlayers = parseInt(document.getElementById('numPlayers').value);

                for (let i = 1; i <= numPlayers; i++) {

                    document.getElementById(`player${i}`).value = '';

                }

                

                vibrate();

            }

        }


        // Handle Enter key in score inputs

        document.addEventListener('keydown', function(e) {

            if (e.key === 'Enter' && e.target.classList.contains('score-input')) {

                e.preventDefault();

                const inputs = Array.from(document.querySelectorAll('.score-input'));

                const currentIndex = inputs.indexOf(e.target);

                

                if (currentIndex < inputs.length - 1) {

                    inputs[currentIndex + 1].focus();

                } else {

                    addManche();

                }

            }

        });
